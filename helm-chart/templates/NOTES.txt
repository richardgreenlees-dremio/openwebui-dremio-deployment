Open WebUI has been deployed!

{{- if eq .Values.openwebui.service.type "LoadBalancer" }}

1. Get the application URL by running these commands:

  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        You can watch the status by running:

  $ kubectl get svc {{ .Values.openwebui.service.name }} -n {{ .Values.namespace }} --watch

  Once the EXTERNAL-IP is available, access Open WebUI at:

  export SERVICE_IP=$(kubectl get svc {{ .Values.openwebui.service.name }} -n {{ .Values.namespace }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo "Open WebUI URL: http://$SERVICE_IP:{{ .Values.openwebui.service.port }}"

{{- else if eq .Values.openwebui.service.type "NodePort" }}

1. Get the application URL by running these commands:

  export NODE_PORT=$(kubectl get svc {{ .Values.openwebui.service.name }} -n {{ .Values.namespace }} -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "Open WebUI URL: http://$NODE_IP:$NODE_PORT"

{{- else if eq .Values.openwebui.service.type "ClusterIP" }}

1. Access Open WebUI by port-forwarding:

  kubectl port-forward -n {{ .Values.namespace }} svc/{{ .Values.openwebui.service.name }} 8080:{{ .Values.openwebui.service.port }}

  Then access Open WebUI at: http://localhost:8080

{{- end }}
{{- end }}
